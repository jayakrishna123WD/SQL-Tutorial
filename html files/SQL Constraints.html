💻 PRIMARY KEY 🌟 PRIMARY KEY = NOT NULL + UNIQUE
🌟 oka user ni unique ga identify cheyyadaniki use ayye constraint ye PRIMARY KEY
🌟 So, primary key constraint ayina column cannot have null values and maintain unique values
🌟 king 👑 yeppudu okkade untadu alane oka table ki oka primary key ne untundi
🌟 ah primary key can consists one or more than one columns
🌟 lets make Student_id column primary key
🍁 Query:

USE JBLCollege;

CREATE TABLE StudentDetails (

Student_id int,

firstname varchar(30),

lastname varchar(30),

Branch text,

CONSTRAINT StudentPK PRIMARY KEY (Student_id)

);

🌟 ee query execute aythe Student_id primary key avtundi ante null values undav and unique values untay
🌟 StudentPK ante Named Constraint, query execute chesaka error vaste ee named constraint display avtundi. meeru multiple tables meeda work chesi nappudu meku ee named constraint baga help avtay because ye table lo error vastundo telustundi. named constraint word yemaina pettukovacchu me istam adhi

💻 FOREIGN KEY 🌟 Tables okadanito okati relation untundi ani cheppa kadha ee topic valle adhi possible avtundi
🌟 we can create a relationship between two tables by using this foreign key constraint
🌟 Foreign key ante table lo oka column that was linked to a table column which has primary key
🌟 simple ga cheppalante primary key unna table parent aythe, foreign key unna table child avtundi

🌟 proper use of foreign keys can minimize data redundancy, incase yeppudaina data change cheyyalanna easy ga untundi access cheyyadaniki
🌟 enti nammatleda oka chinna example cheppukundam, meru college lo chaduvutunnaru anukondi if foreign key concept lekapote department, library, sports tables lo me anni details rayali yemaina change cheyyalanna anni tables lo change cheyyali. same data multiple tables lo undi storage waste.
🌟 if foreignkey concept use cheste, me department table ni sports and library tho link chestaru akkada me id use chestaru simple ga yela aythe me college lo me id number tho anni access cheyyocho alane, ila foreignkey use cheyyadam valla data redundancy ni minimize cheyyochu data fast ga access cheyocchu.

🍁 Query for Parent table:

-- Students Table

CREATE TABLE Students (

Studentid int,

Name varchar(30),

city varchar(30),

Age int,

CONSTRAINT StudentPK PRIMARY KEY (Studentid)

);

🍁 Query for child table:

-- Marks Table

CREATE TABLE Marks (

id int,

Marks varchar(10),

S_id int REFERENCES Students(Studentid),

Result varchar(10),

CONSTRAINT MarksPK PRIMARY KEY (id)

);

🌟 references keyword use chesi foreign keys ni create chestam
🌟 suppose meeru Students and marks table create chesetappudu foreignkey create cheyyaledu marchipoyaru anukondi
🌟 ayina em parledu alter unnadu ga manaki vadni use cheyyadame
🍁 Query:

ALTER TABLE Marks

ADD FOREIGN KEY (S_id) REFERENCES Students(Studentid);

🌟 ila alter command use chesi foreignkey ni create cheyyochu

💻 CHECK Constraint 🌟 nenu netho matladalante oka reason vethukovali, ah reason neeku ok anipiste matladatav. ila conditions unattu, check constraint use chesi ah valid condition ni satisfy chestene data ni teeskuntundi table loki

💻 DEFAULT Constraint 🌟 manam table ki default values pettukovadaniki use chestam ee constraint
🌟 like yelanti situations ante students data tesskunnam anukondi andaridi state AP ne kadha maximum appudu default value AP ista state column ki, AP state kani vallu change chestaru. same ayina vallu malli enter cheyyalsina avasaram undadu.
🍁 Query:

CREATE TABLE students (

userid int PRIMARY KEY,

name varchar(30),

city varchar(25),

state varchar(30) DEFAULT 'AP'

);

🍁 Query:

INSERT INTO students(userid, name, city)

VALUES (1, 'Nagarjuna', 'Vizag');


INSERT INTO students(userid, name, city, state)

VALUES (2, 'Akhil', 'Hyd', 'Telangana');

🍁 Outcome:
userid name city state
1 Nagarjuna vizag AP
2 Akhil Hyd Telangana
🌟 Default using with alter command
🍁 Query:

-- SQL Server

ALTER TABLE students

DEFAULT 'AP' FOR state;


-- PostgreSQL

ALTER TABLE students

ALTER COLUMN state SET DEFAULT 'AP';


-- MySQL

ALTER TABLE students

ALTER state SET DEFAULT 'AP';


-- Oracle

ALTER TABLE students

MODIFY state DEFAULT 'AP';

💻 CREATE INDEX 🌟 yemaina textbook unte open chey starting lo index page untundi dani use meeku telisinde ga manaki kavalsina topic ni fast ga access cheskovadaniki ah index use avtundi
🌟 meeru large amount of data paina work chesetappudu meeku idhi help avtundi
🌟 main purpose of this constraint is, it's faster to retrieve data if we use that column for data retrieval
🍁 Query:

-- Creating a table

CREATE TABLE Books (

Booksid int PRIMARY KEY,

Name varchar(40),

Author varchar(35),

Price int NOT NULL

);


-- creating index for column Name

CREATE INDEX BookIndex

ON Books(Name);

🌟 For example bank database teeskondi andulo lakhs of people data untundi, akkada accessing easy avvadaniki ee index use chestaru
🌟 but same name chala mandiki untundi ga appudu yela retrieve avtundi, manaki unique unnadu ga multiple columns ki kooda index create cheyyochu, first and last name kalipi index chesi unique cheste appudu unique ga identification easy avtundi
🍁 Query:

-- Creating a table

CREATE TABLE Bank (

AccountNumber int PRIMARY KEY,

firstname varchar(40),

lastname varchar(35),

);


-- creating index for column Name

CREATE UNIQUE INDEX BankIndex

ON Bank(firstname, lastname);

🌟 maku index em vaddhu, danni teeseyalante

--SQL Server

DROP INDEX Bank.BankIndex;


-- PostgreSQL, Oracle

DROP INDEX BankIndex;


-- MySQL

ALTER TABLE Bank

DROP INDEX BankIndex;
