💻 CREATE DATABASE Statement 🌟 Multiple Statements ni semicolon seperate chestundi, konni RDBMS softwares lo pettakapoyina parledu. For good practice semicolon pedithe better.
🌟 SQL is not case sensitive, kabatti uppercase or lowercase yedi use chesina parledu but for code readability we mention all commands, functions, Statements, logical operators, joins, etc.., in uppercase

CREATE DATABASE Manade_idanta;

🍁 CREATE command use chesi Manade_idanta ane database ni create chesam indulo tables store cheskovacchu manam

🌟 okavela Manade_idanta database, software lo munde unte manaki error vastundi because already ah database undi kadha anduke.
🌟 IF NOT EXISTS, ee command em chestundante first mention chesina name tho yemaina databases unnayo ledo choostundi, if lekapote create chestundi unte neglect chestundi.

CREATE DATABASE IF NOT EXISTS Manade_idanta;

🌟 manam yelano create chesasam ga, so malli create cheyyadu.
🌟 Ippudu manam software lo yennni databases unnayo telusukovalante SHOW command use chestam

SHOW DATABASES;

🌟 Two databases unnay meeru first one lo unna tables ni access cheyyali, appudu manam USE statement ni use cheyyali

USE Manade_idanta;

🌟 Ippudu Manade_idanta database lo unna tables ni access cheyyochu, if second database name Books anukondi danini use cheyyalante USE pakkana name ni replace chestam ila databases switch avvochu

USE Books;

💻 CREATE TABLE Statement 🌟 create table statement use chesi database lo table ni create chestam
🌟 Let's create table which stores user details

USE Manade_idanta;

CREATE TABLE UserDetails (

id int,

firstname varchar(30),

lastname varchar(30),

city text,

phone varchar(10)

);

🌟 ee query ni execute cheste Manade_idanta database lo UserDetails table create avtundi, manam table lo columns id, firstname, lastname, city, phone ni create chesam. Visit Datatypes for clarification
🌟 using id we can identify users uniquely because that was unique for every user, we'll discuss about that in comming sessions
🌟 if UserDetails tho table already unte error vastundi, deeniki solution ye IF NOT EXISTS command.

USE Manade_idanta;

CREATE TABLE IF NOT EXISTS ComedyMovies (

id int,

MovieName varchar(30),

ReleaseDate int

);

🌟 ComedyMovies tho ye table ledu kabatti query execute avtundi table create avtundi

💻 DROP DATABASE Command 🌟 sonusood feels like oka pakka storage leka nenunte nuvvu extra ga database lu create chestunnava
🌟 Me : anduke database delete chesesa oh meeku telidu kadha yela delete chesano let's see

DROP DATABASE Books;

🌟 ee query execute aythe Books lo unna tables and total Books database delete ayipotundi
🌟 Only table okate delete cheyyali ante Drop table command use chestam

DROP TABLE ComedyMovies;

🌟 ComedyMovies table delete ayipotundi.
🌟 Database lo leni tables ni delete cheste error vastundi, solution is meeku telusu by using IF EXISTS command

DROP TABLE IF EXISTS ComedyMovies;

🌟 ee table already delete ayipoyindi kadha. So, query execute ayina if EXISTS command valla error raadu

💻 ALTER TABLE Statement 🌟 ee command use chesi columns ni add, modify, rename and delete cheyocchu & table name kooda change cheyocchu
🌟 Adding new column into a table

-- adding single column

ALTER TABLE UserDetails

ADD pincode varchar(7);

-- adding more than one column at a time

ALTER TABLE UserDetails

ADD pincode varchar(7), TokenNumber int;

🌟 column name changing in table

ALTER TABLE UserDetails

RENAME COLUMN TokenNumber TO T_num;

🌟 I want to increase the length of firstname in UserDetails table. we use modify or alter column command, kaani different RDBMS lo different syntax untundi lets see

-- SQL Server

ALTER TABLE UserDetails

ALTER COLUMN firstname varchar(25);


-- MySQL

ALTER TABLE UserDetails

MODIFY COLUMN firstname varchar(25);


-- Oracle

ALTER TABLE UserDetails

MODIFY firstname varchar(25);


-- PostgreSQL

ALTER TABLE UserDetails

ALTER COLUMN firstname TYPE varchar(25);

🌟 Delete a column in table

ALTER TABLE UserDetails

DROP COLUMN T_num;

🌟 Delete multiple column in table

-- SQL Server

ALTER TABLE UserDetails

DROP COLUMN T_num, pincode, phone;


-- MySQL

ALTER TABLE UserDetails

DROP T_num, DROP pincode, DROP phone;


-- Oracle

ALTER TABLE UserDetails

DROP (T_num, pincode, phone);


-- PostgreSQL

ALTER TABLE UserDetails

DROP COLUMN T_num, DROP COLUMN pincode, DROP COLUMN phone;

🌟 changing the name of table

ALTER TABLE UserDetails

RENAME TO YourNameDetails;

💻 Truncate Table 🌟 Table lo unna rows anni delete ayipovali but table matram database lo undalante truncate command use chestam table lo data antha delete chestundi but table aythe database lo untundi

TRUNCATE TABLE YourNameDetails;

🌟 YourNameDetails table aythe untundi kani data em undadu
